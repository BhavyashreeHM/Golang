package main

import (
	"fmt"
	"time"
)

func main() {
	c := make(chan string)
	go greet(c, "HELLO")
	go greet(c, "WORLD")
	fmt.Println("Greeting", <-c)
	// time.Sleep(time.Microsecond * 100)

}

func greet(ch chan string, name string) {
	for _, n := range name {
		fmt.Println(string(n), "")
	time.Sleep(time.Microsecond * 100)
		
	}
	ch <- name
	
}
