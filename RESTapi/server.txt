package main

import (
	"fmt"
	"log"
	"net/http"
)

func main() {
	const serAddr string = "127.0.0.1:3001"
	println("Server is running at PORT", serAddr)

	// Server code would go here
	err := http.ListenAndServe(serAddr, nil)
	if err != nil {
		// println("Error starting server:", err.Error())
		log.Fatal("Error starting server:", err)
	}

	http.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
		// w.Write([]byte("Hello, World!"))
		fmt.Fprintln(w, "Hello, Server!")
	})
}
